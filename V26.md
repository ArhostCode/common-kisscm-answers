 ```bash
#!/bin/bash

# Проверяем, что передан аргумент
if [ -z "$1" ]; then
  echo "Usage: $0 <directory>"
  exit 1
fi

# Переходим в указанную директорию
cd "$1" || exit

# Ищем файлы с расширением .jpg и содержащие дату 2022 года в названии
find . -type f -name "*.jpg" | grep -E "[0-9]{2}-[0-9]{2}-2022"
```
